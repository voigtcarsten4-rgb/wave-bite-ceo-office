🔧 FFmpeg Batch Commands

# Resize für alle Plattformen gleichzeitig ffmpeg -i input.mp4 \ -vf "scale=1080:1920" output_9x16.mp4 \ -vf "scale=1080:1080" output_1x1.mp4 \ -vf "scale=1920:1080" output_16x9.mp4 # Voiceover + Musik mischen (Musik leiser) ffmpeg -i video.mp4 -i voiceover.mp3 -i music.mp3 \ -filter_complex "[2:a]volume=0.3[bg]; [1:a][bg]amix=inputs=2[out]" \ -map 0:v -map "[out]" output.mp4 # Auto-Untertitel hinzufügen ffmpeg -i video.mp4 -vf "subtitles=subs.srt" output_sub.mp4 # Batch Export alle Videos im Ordner for f in /raw/*.mp4; do ffmpeg -i "$f" -vf "scale=1080:1920,setsar=1" \ /export/reels/"${f%.mp4}_reel.mp4" done

🚀 Setup-Anleitung: Runway ML ▼

1. Account erstellen: runwayml.com → Pro Plan empfohlen (500 Credits/Monat)

2. API-Key generieren: Settings → API → New Token → In .env speichern

3. Empfohlenes Modell: Gen-3 Alpha Turbo für schnelle Iterationen, Gen-3 Alpha für finale Qualität

4. Workflow: Text → Image (Leonardo) → Image-to-Video (Runway) für beste Resultate

5. Wave Bite Settings: Immer "cinematic", "photorealistic", "no text overlays" in System-Prompt einfügen

🎙️ Setup-Anleitung: ElevenLabs ▼

1. Account: elevenlabs.io → Creator Plan (22.000 Zeichen/Monat)

2. Empfohlene Stimme: Adam (en-US) oder Daniel (en-GB) für Premium-Feeling

3. Voice Settings: Stability 0.65 | Clarity 0.80 | Style 0.20

4. API Integration via Make.com: Text-Input → ElevenLabs API → MP3-Output → Cloudinary

5. Deutsche Stimme: "Antoni" oder "Matei" für DE-Märkte

🎵 Setup-Anleitung: Suno AI ▼

1. Account: suno.ai → Pro Plan (2.500 Credits/Monat)

2. Modell: Suno v4 für beste Qualität

3. Wave Bite Standard-Style: "cinematic ambient electronic, instrumental, no vocals, maritime atmosphere, 85-90 BPM"

4. Custom Mode nutzen für präzise Kontrolle

5. Alle generierten Tracks in Cloudinary speichern mit Tags: [campaing-name] [format] [mood]

Wave Bite — Remotion Starter Komponente

// wavebite-reel-intro.tsx // Automatisches Branded Intro für alle Wave Bite Reels import { AbsoluteFill, Sequence, useCurrentFrame, interpolate, spring } from 'remotion'; const BRAND_TEAL = '#0E9E8A'; const BRAND_GOLD = '#C9A84C'; const BRAND_DARK = '#070B14'; export const WaveBiteIntro: React.FC<{ title: string; subtitle?: string; duration?: number; }> = ({ title, subtitle, duration = 90 }) => { const frame = useCurrentFrame(); // Logo fade in const logoOpacity = interpolate(frame, [0, 20], [0, 1], { extrapolateRight: 'clamp' }); const logoScale = spring({ frame, fps: 30, config: { damping: 12 } }); // Title slide in const titleY = interpolate(frame, [15, 40], [30, 0], { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }); const titleOpacity = interpolate(frame, [15, 40], [0, 1], { extrapolateRight: 'clamp' }); // Wave animation (CSS-based) const waveOffset = interpolate(frame, [0, duration], [0, -200]); return ( <AbsoluteFill style={{ background: BRAND_DARK, fontFamily: 'Segoe UI, system-ui' }}> {/* Animated Wave Background */} <div style={{ position: 'absolute', bottom: 0, width: '120%', left: '-10%', height: '40%', opacity: 0.15, background: `linear-gradient(180deg, transparent, ${BRAND_TEAL})`, transform: `translateX(${waveOffset}px)`, borderRadius: '50% 50% 0 0 / 20% 20% 0 0', }}/> {/* Gold accent line */} <div style={{ position: 'absolute', top: '50%', left: 0, right: 0, height: '1px', background: `linear-gradient(90deg, transparent, ${BRAND_GOLD}, transparent)`, opacity: logoOpacity, }}/> {/* Logo + Brand */} <AbsoluteFill style={{ justifyContent: 'center', alignItems: 'center', flexDirection: 'column', gap: 16 }}> <div style={{ fontSize: 64, transform: `scale(${logoScale})`, opacity: logoOpacity, }}>🌊</div> <div style={{ fontSize: 32, fontWeight: 800, color: 'white', opacity: titleOpacity, transform: `translateY(${titleY}px)`, }}> <span style={{ color: BRAND_GOLD }}>Wave</span> Bite </div> {title && <div style={{ fontSize: 18, color: BRAND_TEAL, fontWeight: 500, opacity: titleOpacity, transform: `translateY(${titleY}px)`, letterSpacing: '2px', textTransform: 'uppercase', }}>{title}</div>} </AbsoluteFill> </AbsoluteFill> ); }; // Composition registrieren: // registerRoot(() => <Composition id="WaveBiteIntro" component={WaveBiteIntro} // durationInFrames={90} fps={30} width={1080} height={1920} // defaultProps={{ title: 'Neues Feature', subtitle: 'Wave Bite App' }} />)

Batch Render Script

// render-batch.mjs // Rendert alle Kampagnen-Videos auf einmal import { renderMedia, selectComposition } from '@remotion/renderer'; import { campaigns } from './campaigns.json'; for (const campaign of campaigns) { const composition = await selectComposition({ serveUrl: 'http://localhost:3000', id: 'WaveBiteReel', inputProps: campaign, }); await renderMedia({ composition, serveUrl: 'http://localhost:3000', codec: 'h264', outputLocation: `./output/${campaign.id}.mp4`, }); console.log(`✓ Rendered: ${campaign.id}`); }

Bria AI — fal.ai API Integration

// product-shot-generator.js // Sendet Produktbild → erhält Lifestyle-Shot import * as fal from "@fal-ai/serverless-client"; fal.config({ credentials: process.env.FAL_KEY }); async function generateProductShot(productImageUrl, scene) { const result = await fal.subscribe( "fal-ai/bria/product-shot", { input: { image_url: productImageUrl, scene_description: scene, // Wave Bite Scenes: // "Premium marina at golden hour, luxury boats background" // "Wooden boat deck, calm water, sunset lighting" // "Waterfront restaurant table, teal water background" num_images: 4, sync_mode: true, }, logs: true, }); return result.images; // Array von generierten Bildern } // Beispiel-Aufruf: generateProductShot( "https://cdn.wavebite.de/bunbo-burger.jpg", "Luxury wooden boat deck, golden hour sunset, calm river water in background, premium lifestyle photography" );
